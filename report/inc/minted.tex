%% Pre-defined syntax
% py  -> python
% c   -> c
% vl  -> verilog
% sc  -> scala
% mat -> matlab
% log -> console

%% suffices
% inl     -> inline
% box     -> background, titled
% file    -> background, no title
% filebox -> background, titled

% -------- Example --------
\iffalse
    \begin{pybox}{Title}
        code here
    \end{pybox}

    \pyinl{a==1}

    \pyfile{a.py}

    \pyfilebox[Title]{a.py}

    \begin{minted}{python}
        code here
    \end{minted}
\fi
% ------------------------

\usepackage[skins,breakable,minted]{tcolorbox}
